# Set Clothing Rules
sub_make_nude_interaction = {
	category = interaction_category_friendly
	common_interaction = yes
	interface_priority = 101

	desc = sub_make_nude_interaction_desc

	auto_accept = yes

	is_shown = {
		scope:recipient = {
			AND = {
				NOT = { has_trait = sub_clothing_rule_3 }
				sub_show_sub_interactions_trigger = yes
			}
		}
	}

	is_valid_showing_failures_only = {
		scope:recipient = {
			sub_has_obedience_modifier_trigger = yes
		}
	}

	on_accept = {
		scope:recipient = {
			add_trait = sub_clothing_rule_3
			add_opinion = {
				modifier = sub_made_me_go_naked
				target = scope:actor
			}
		}

		scope:actor = {
			send_interface_toast = {
				title = sub_accept_order
				right_icon = scope:recipient
			}
		}
	}
}

# Start sub_train_obedience scheme
sub_train_obedience_interaction = {
	category = interaction_category_friendly

	common_interaction = yes
	interface_priority = 150

	auto_accept = yes
	ignores_pending_interaction_block = yes

	scheme = sub_train_obedience

	send_name = START_SUB_TRAINING

	desc = sub_train_obedience_interaction_desc

	is_shown = {
		scope:recipient = {
			sub_can_be_obedience_trained_interaction_trigger = yes
		}
	}

	is_valid = {
		scope:actor = { is_ai = no }
	}

	is_valid_showing_failures_only = {
		scope:actor = {
			is_adult = yes
			NOT = {
				is_scheming_against = { target = scope:recipient }
			}
		}
	}

	on_accept = {
		scope:actor = {
			hidden_effect = {
				send_interface_toast = {
					title = sub_train_obedience_interaction_toast
					left_icon = scope:actor
					right_icon = scope:recipient

					start_scheme = {
						target = scope:recipient
						type = sub_train_obedience
					}
				}
			}
		}
	}
}
