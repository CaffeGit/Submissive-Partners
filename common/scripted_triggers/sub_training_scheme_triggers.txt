sub_can_be_trained_scheme_trigger = {
	AND = {
		is_adult = yes
		is_female = yes
		OR = {
			is_consort_of = scope:owner
			has_relation_lover = scope:owner
			has_relation_soulmate = scope:owner
			has_relation_dominant = scope:owner
		}
	}
}

sub_can_be_obedience_trained_scheme_trigger = {
	AND = {
		NOT = { has_trait = sub_subservient_4 }
		NOT = {
			has_opinion_modifier = {
				target = scope:owner
				modifier = sub_subservient_to_master
			}
		}
		sub_can_be_trained_scheme_trigger = yes
	}
}
